### ImplementaĂ§ĂŁo de `calculate()` com PrecisĂŁo Decimal

```javascript
function calculate() {
  // RepetiĂ§ĂŁo de = consecutivo
  if (state.operator === null && state.lastOperator !== null) {
    state.previousValue = state.currentValue;
    state.operator = state.lastOperator;
    state.currentValue = state.lastOperand;
  }

  if (state.operator === null || state.previousValue === '') return;

  const prev = parseFloat(state.previousValue);
  const curr = parseFloat(state.currentValue);

  // Guard: divisĂŁo por zero
  if (state.operator === 'Ă·' && curr === 0) {
    state.currentValue = 'ERROR! đź’Ą';
    state.operator = null;
    state.previousValue = '';
    state.shouldResetDisplay = true;
    updateDisplay();
    return;
  }

  let result;
  switch (state.operator) {
    case '+': result = prev + curr; break;
    case '-': result = prev - curr; break;
    case 'Ă—': result = prev * curr; break;
    case 'Ă·': result = prev / curr; break;
    default: return;
  }

  // Salvar para repetiĂ§ĂŁo do = ANTES de resetar
  state.lastOperator = state.operator;
  state.lastOperand = state.currentValue;

  // CRĂŤTICO: normalizar ponto flutuante

  state.operator = null;
  state.previousValue = '';
  state.shouldResetDisplay = true;

  updateDisplay();
}
```

### Event Delegation Pattern

```javascript
document.addEventListener('DOMContentLoaded', () => {
  const buttonsContainer = document.querySelector('.buttons');

  buttonsContainer.addEventListener('click', (e) => {
```
